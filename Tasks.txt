3)	Создать группу через Django shell. В группу добавить возможность добавлять изменять
удалять и просматривать владельцев отелей. Через shell,
наделить правами этой группы одного из пользователей, созданных в первом задании.

hotelowner_group = Group.objects.create(name="Hotel owner group")
add_hotelowner = Permission.objects.get(codename="add_hotelowner")
change_hotelowner = Permission.objects.get(codename="change_hotelowner")
delete_hotelowner = Permission.objects.get(codename='delete_hotelowner')
view_hotelowner = Permission.objects.get(codename="view_hotelowner")
hotelowner_group.permissions.add(add_hotelowner, change_hotelowner, delete_hotelowner, view_hotelowner)
user = User.objects.get(username="Mihail")
user.groups.add(hotelowner_group)
user.save()
